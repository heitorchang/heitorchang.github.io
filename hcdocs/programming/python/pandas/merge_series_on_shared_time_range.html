Given a common set of times, merge multiple series with incomplete time data.

>>> import pandas as pd
>>> from functools import reduce
>>> from datetime import datetime
>>> full_time_datetimeindex = pd.date_range('2003-01-01', '2005-01-01', freq='MS')
>>> df_full_time = full_time_datetimeindex.to_frame()
>>> series1 = pd.Series([1, 3, 5], index=[datetime(2003, 1, 1), datetime(2003, 2, 1), datetime(2004, 10, 1)], name='s1')
>>> series2 = pd.Series([20, 22, 25], index=[datetime(2004, 2, 1), datetime(2004, 5, 1), datetime(2004, 9, 1)], name='s2')

>>> data_series = [df_full_time, series1, series2]
>>> df_all = reduce(lambda left, right: pd.merge(left, right, left_index=True, right_index=True, how='outer'), data_series)

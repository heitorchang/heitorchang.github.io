Command-line script (cli-utils)

The real advantage of this script (over a Bash or Python script) is the clean separation of Error, standard out, and return code of the program execution.

Use :output t, :error-output t to redirect the streams (instead of saving in a string).

ASDF definition

(defpackage :cli-utils
  (:use :asdf :cl)
  (:export
   :program-out-err))

(in-package :cli-utils)

(defsystem "cli-utils"
  :description "CLI Utils"
  :version "0.0.1"
  :author "Heitor Chang"
  :license "MIT"
  :components ((:file "cli-utils")))

Helper

(in-package :cli-utils)

(defun program-out-err (program-list)
  (multiple-value-bind (out err return-code)
      (uiop:run-program program-list
                        :output :string
                        :error-output :string
                        :ignore-error-status t)
    (format t "PROGRAM: ~A~%" program-list)
    (when err (format t "ERROR: ~A~%~%" err))
    (format t "OUT:~%~A~%" out)
    (format t "RETURN CODE: ~A~%" return-code)))

Usage

#!/usr/bin/env -S sbcl --script

(require 'asdf)
(asdf:load-system "cli-utils")

(let ((args (uiop:command-line-arguments)))
  (cli-utils:program-out-err (list "echo" (first args))))

#!/usr/bin/env -S sbcl --script

(require 'asdf)
(asdf:load-system "cli-utils")

(let ((args (uiop:command-line-arguments)))
  (cli-utils:program-out-err `("echo" ,(first args) ,@ (subseq args 2))))
